---ELIMINAR EQUIPOS---
DELETE FROM "CPSAA"."GESAC_MAE_EQUI" WHERE "SCMEQ_IDE_EQUI_K" IN (SELECT  equi."SCMEQ_IDE_EQUI_K" FROM "CPSAA"."GESAC_MAE_EQUI" equi WHERE equi."SCMPC_IDE_PROC_K" IN (SELECT PROC1."SCMPC_IDE_PROC_K" FROM "CPSAA"."GESAC_MAE_PROC"  PROC1 INNER JOIN "CPSAA"."GESAC_MAE_PROC" PROC2 ON (PROC1."SCMPC_IDE_PROC_BASE_K"=PROC2."SCMPC_IDE_PROC_K") AND PROC2."SCMPC_NOM_PROC"='TGA' AND PROC1."SCMPC_COC_EST"='ACT' AND PROC2."SCMPC_COC_EST"='ACT'));
---ELIMINAR PROCESO---
DELETE FROM "CPSAA"."GESAC_MAE_PROC" WHERE "SCMPC_IDE_PROC_BASE_K" IN (SELECT proc."SCMPC_IDE_PROC_K" FROM "CPSAA"."GESAC_MAE_PROC" proc WHERE proc."SCMPC_NOM_PROC" = 'TGA') AND "SCMAP_IDE_PLAN_K" IN (SELECT plan."SCMAP_IDE_PLAN_K" FROM "CPSAA"."GESAC_MAE_PLAN" plan WHERE plan."SCMAP_NOM_PLAN" = 'Muestras de Investigación');
DELETE FROM "CPSAA"."GESAC_MAE_PROC" WHERE "SCMPC_IDE_PROC_K" IN (SELECT proc."SCMPC_IDE_PROC_K" FROM "CPSAA"."GESAC_MAE_PROC" proc WHERE proc."SCMPC_NOM_PROC" = 'TGA');
